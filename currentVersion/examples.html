Example 01 This is a basic example for Liefs Layout Manager.

This page (see view source this page) demonstrates a simple layout.
It consists of x2 DIV tags: id="Menu" and id="right"

This is the COMPLETE Javascript Code required for Liefs Layout Manager to Map this Page. 
h("example01", 4 ,
  I("Menu","200px", "150px", "250px"),
  I("ex01", "100%")
);
line:1 - Lief's Layout Manager function "h" defines a Horizontal Item, named "Master" with 4px Margins.

line:2 - Lief's Layout Manager function "I" defines an Item, named "Menu". For this statement to be valid, there must be exactly one matching selector (default is id ...so matching id) or an error will be thrown. This "Menu" Item will be 400px Wide, complete with dragBar stretching to a minimum value of 300px to a maximum value of 500px.

line:3 - Lief's Layout Manager function "I" defines an Item, named "right". For this statement to be valid, there must be exactly one matching selector (default is id ...so matching id) or an error will be thrown. This "right" Item will be 100% of the pixels left, after all others are subtracted from available.

line:4 close brackets, end line

Note: Liefs Layout Manager must be loaded with a script tag. 
<script src="...js">
Note: Liefs Layout Manager REQUIRES basic css styling of dragBars. Customize to suit, but they must be position: fixed to work. 
  .Hdragbar {position: fixed;cursor: col-resize;border-style: none;border-width: 0px;border-radius: 0px;}
  .Vdragbar {position: fixed;cursor: row-resize;border-style: none;border-width: 0px;border-radius: 0px;}

Example 03 - Layouts Are Managed By Conditional Functions

This Example shows 4 Different Items, 4 Different Ways.

In this example, we created 4 Layous, each dependant on a conditional value. In this case, the value of "ex3Layout" - which we created to suit our needs. 
<button onclick="ex3Layout=0;Handler.resizeEvent();">1</button>
<button onclick="ex3Layout=1;Handler.resizeEvent();">2</button>
<button onclick="ex3Layout=2;Handler.resizeEvent();">3</button>
<button onclick="ex3Layout=3;Handler.resizeEvent();">4</button>

var ex3Layout=0;
var MenuR = I("Menu","100px", "100px", "250px");
var ex3col = h("ex3col", 8 ,
                v("100%", 8,
                    I("ex03", "50%"),
                    h("50%", 8,
                      I("I01", "25%"),
                      I("I02", "25%"),
                      I("I03", "25%"),
                      I("I04", "25%")
                    )
                ),
                 MenuR
               );
 var ex3row = h("ex3row", 8 ,
                 v("100%", 8,
                     I("ex03", "50%"),
                     I("I01", "12%"),
                     I("I02", "13%"),
                     I("I03", "12%"),
                     I("I04", "13%")
                 ),
                  MenuR
                );

  var ex3grid = h("ex3grid", 8 ,
                  v("100%", 8,
                      I("ex03", "50%"),
                      h("25%", 8,
                        I("I01", "50%"),
                        I("I02", "50%")
                      ),
                      h("25%", 8,
                        I("I03", "50%"),
                        I("I04", "50%")
                      )
                  ),
                   MenuR
                 );
   var ex3cust = h("ex3cust", 8,
                   v("100%", 8,
                       I("ex03", "50%"),
                       I("I01", "25%"),
                       h("25%", 8,
                         I("I02", "33%"),
                         I("I03", "33%"),
                         I("I04", "34%")
                       )
                   ),
                    MenuR
                  );
  var ex3L1 = L("ex03L1", ex3col, function () { return (ex3Layout === 0); });   // define layouts
  var ex3L2 = L("ex03L2", ex3row, function () { return (ex3Layout === 1); });   // define layouts
  var ex3L3 = L("ex03L3", ex3grid, function () { return (ex3Layout === 2); });  // define layouts
  var ex3L4 = L("ex03L4", ex3cust, function () { return (ex3Layout === 3); });  // define layouts

  H("Main", [ex3L1, ex3L2, ex3L3, ex3L4] ); // hander given 4 layous each with conditional function
