Example 03 - Layouts Are Managed By Conditional Functions

This Example shows 4 Different Items, 4 Different Ways.

In this example, we created 4 Layous, each dependant on a conditional value. In this case, the value of "ex3Layout" - which we created to suit our needs. 
<button onclick="ex3Layout=0;Handler.resizeEvent();">1</button>
<button onclick="ex3Layout=1;Handler.resizeEvent();">2</button>
<button onclick="ex3Layout=2;Handler.resizeEvent();">3</button>
<button onclick="ex3Layout=3;Handler.resizeEvent();">4</button>

var ex3Layout=0;
var MenuR = I("Menu","100px", "100px", "250px");
var ex3col = h("ex3col", 8 ,
                v("100%", 8,
                    I("ex03", "50%"),
                    h("50%", 8,
                      I("I01", "25%"),
                      I("I02", "25%"),
                      I("I03", "25%"),
                      I("I04", "25%")
                    )
                ),
                 MenuR
               );
 var ex3row = h("ex3row", 8 ,
                 v("100%", 8,
                     I("ex03", "50%"),
                     I("I01", "12%"),
                     I("I02", "13%"),
                     I("I03", "12%"),
                     I("I04", "13%")
                 ),
                  MenuR
                );

  var ex3grid = h("ex3grid", 8 ,
                  v("100%", 8,
                      I("ex03", "50%"),
                      h("25%", 8,
                        I("I01", "50%"),
                        I("I02", "50%")
                      ),
                      h("25%", 8,
                        I("I03", "50%"),
                        I("I04", "50%")
                      )
                  ),
                   MenuR
                 );
   var ex3cust = h("ex3cust", 8,
                   v("100%", 8,
                       I("ex03", "50%"),
                       I("I01", "25%"),
                       h("25%", 8,
                         I("I02", "33%"),
                         I("I03", "33%"),
                         I("I04", "34%")
                       )
                   ),
                    MenuR
                  );
  var ex3L1 = L("ex03L1", ex3col, function () { return (ex3Layout === 0); });   // define layouts
  var ex3L2 = L("ex03L2", ex3row, function () { return (ex3Layout === 1); });   // define layouts
  var ex3L3 = L("ex03L3", ex3grid, function () { return (ex3Layout === 2); });  // define layouts
  var ex3L4 = L("ex03L4", ex3cust, function () { return (ex3Layout === 3); });  // define layouts

  H("Main", [ex3L1, ex3L2, ex3L3, ex3L4] ); // hander given 4 layous each with conditional function
